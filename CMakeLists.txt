# Partial (add_subdirectory only) and experimental CMake support
# Subject to change; please do not rely on the contents of this file yet.
# This file is NOT supposed to be used to build the library, use Boost.Build for that.

cmake_minimum_required(VERSION 3.5)

project(BoostSync LANGUAGES CXX)

add_library(boost_sync INTERFACE)
add_library(Boost::sync ALIAS boost_sync)

target_include_directories(boost_sync INTERFACE include)

target_link_libraries(boost_sync
    INTERFACE
        Boost::assert
        Boost::atomic
        Boost::chrono
        Boost::config
        Boost::core
        Boost::date_time
        Boost::move
        Boost::mpl
        Boost::preprocessor
        Boost::ratio
        Boost::static_assert
        Boost::system
        Boost::throw_exception
        Boost::winapi
)
